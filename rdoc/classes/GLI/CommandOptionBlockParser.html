def do_parse(args)
  unknown_options = []
  @option_parser_factory.option_parser.order!(args) do |non_option|
    unknown_options << non_option
    break if break_on_non_option?
  end
  unknown_options.reverse.each do |unknown_option|
    args.unshift(unknown_option)
  end
  args
end
